	${PROJECT_NAME} PUBLIC

)	
	
#Define the project include directories

	${PROJECT_NAME} PUBLIC
	#Header path used at build time
	$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>  
	
	#Header path used when installed
	$<INSTALL_INTERFACE:include>
)

  ${PROJECT_NAME} 
  GTest::gtest
  bofstd
  ${BLD_STANDARD_LINK_LIBRARIES}
)
